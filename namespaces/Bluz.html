Validator¶

Example of usage use Bluz\Validator\Traits\Validator; use Bluz\Validator\Validator as v;

class Row extends Db\Row {

   use Validator;
   function beforeSave()
   {
        $this->addValidator(
            'login',
            v::required()->latin()->length(3, 255)
        );
   }

}

    'memcached' => [],       // Settings for another cache adapter
