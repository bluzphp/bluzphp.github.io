/**

namespace Bluz\Grid\Helper;

use Bluz\Grid;

return
    /**
     * @return string|null $url
     */
    function () {
        /**
         * @var Grid\Grid $this
         */
        if ($this->getPage() <= 1) {
            return null;
        }

        return $this->getUrl(['page' => $this->getPage() - 1]);
    };
