final class Response
{
    use ProxyTrait;

    /**
     * Init instance
     *
     * @throws ComponentException
     */
    protected static function initInstance()
    {
        throw new ComponentException("Class `Proxy\\Request` required external initialization");
    }

    /**
     * Redirect to URL
     *
     * @param  string $url
     *
     * @return void
     * @throws RedirectException
     */
    public static function redirect($url)
    {
        $redirect = new RedirectException();
        $redirect->setUrl($url);
        throw $redirect;
    }

    /**
     * Redirect to controller
     *
     * @param  string $module
     * @param  string $controller
     * @param  array  $params
     *
     * @return void
     * @throws RedirectException
     */
    public static function redirectTo($module = 'index', $controller = 'index', array $params = [])
    {
        $url = Router::getUrl($module, $controller, $params);
        self::redirect($url);
    }

    /**
     * Reload current page please, be careful to avoid loop of reload
     *
     * @return void
     * @throws RedirectException
     */
    public static function reload()
    {
        self::redirect(Request::getUri());
    }
}
