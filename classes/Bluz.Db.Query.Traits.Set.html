set(string $key, string $value) : \Bluz\Db\Query\Insert|\Bluz\Db\Query\Update

Sets a new value for a column in a insert/update query

$ub = new UpdateBuilder();
$ub
    ->update('users')
    ->set('password', md5('password'))
    ->where('id = ?');
